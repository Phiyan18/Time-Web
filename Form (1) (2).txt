using System;
using System.Collections.Generic;
using System.Web.UI;

namespace EmployeeFormWebApp
{
    public partial class Default : Page
    {
        private static List<Employee> employeeList = new List<Employee>();

        protected void Page_Load(object sender, EventArgs e)
        {
            if (!IsPostBack)
            {
                BindGrid();
            }
        }

        protected void btnInsert_Click(object sender, EventArgs e)
        {
            Employee emp = new Employee
            {
                ID = txtEmpID.Text.Trim(),
                Name = txtEmpName.Text.Trim(),
                City = ddlCity.SelectedValue,
                Gender = rdoMale.Checked ? "Male" : "Female"
            };

            employeeList.Add(emp);
            BindGrid();
            ClearFields();
        }

        protected void btnDelete_Click(object sender, EventArgs e)
        {
            if (gvEmployees.SelectedIndex >= 0)
            {
                string id = gvEmployees.SelectedDataKey.Value.ToString();
                var emp = employeeList.Find(e => e.ID == id);
                if (emp != null)
                {
                    employeeList.Remove(emp);
                    BindGrid();
                }
            }
        }

        private void BindGrid()
        {
            gvEmployees.DataSource = employeeList;
            gvEmployees.DataBind();
        }

        private void ClearFields()
        {
            txtEmpID.Text = "";
            txtEmpName.Text = "";
            ddlCity.SelectedIndex = 0;
            rdoMale.Checked = true;
            rdoFemale.Checked = false;
        }

        protected void gvEmployees_SelectedIndexChanged(object sender, EventArgs e)
        {
            // Row is selected for delete
        }
    }

    public class Employee
    {
        public string ID { get; set; }
        public string Name { get; set; }
        public string City { get; set; }
        public string Gender { get; set; }
    }
}
